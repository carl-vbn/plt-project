# Very basic example
Canvas(width: 50, height: 50, length: 10)
    Fill(Solid('black'))
        Rectangle(x: 0, y: 0, width: 50, height: 50)

    Fill(Solid('red'))
        Stroke(Solid('white'), width: 2)
            Rectangle(x: 0, y: 0, width: 25, height: 25)

    Fill(Solid('blue'))
        Rectangle(x: 25, y: 25, width: 25, height: 25)
