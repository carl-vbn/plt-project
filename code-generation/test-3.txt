# Overlapping Rectangles
Canvas(width: 100, height: 100) # 100x100 px canvas
    # A small black rectangle
    Fill(Solid('black'))
        Rectangle(x: 0, y: 0, width: 100, height: 100) 

    # A larger, offset blue square overlapping with the above
    Fill(Solid('blue'))
        Rectangle(x: 0, y: 0, width: 50, height: 50)
    
    # A LaTeX label
    Fill(Solid('white'))
        Latex('\textbf{Overlapping Rectangles}', x: 20, y: 25)
